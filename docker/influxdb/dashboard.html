Sample Queries

Recent Webhook Activity (Last Hour)

from(bucket: "mai_usage_raw_dev")
  |> range(start: -1h)
  |> filter(fn: (r) => r._measurement == "usage_tracking")
  |> sort(columns: ["_time"], desc: true)
  |> limit(n: 50)

Daily Token Usage by Model

from(bucket: "mai_usage_raw_dev")
  |> range(start: -7d)
  |> filter(fn: (r) => r._measurement == "usage_tracking")
  |> filter(fn: (r) => r._field == "total_tokens")
  |> group(columns: ["model"])
  |> aggregateWindow(every: 1d, fn: sum)

Cost Analysis with Markup

from(bucket: "mai_usage_raw_dev")
  |> range(start: -30d)
  |> filter(fn: (r) => r._measurement == "usage_tracking")
  |> filter(fn: (r) => r._field == "cost_usd")
  |> group(columns: ["client_org_id"])
  |> sum()
  |> map(fn: (r) => ({ r with _value: r._value * 1.3 }))
